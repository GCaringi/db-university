SELECT 	
    COUNT(`id`) AS "Numero studenti",
	YEAR(`enrolment_date`) AS "Anno d'iscrizione"
FROM 
    students
GROUP BY YEAR(`enrolment_date`);


2.
SELECT 
	COUNT(`id`) AS "Numero di professori",
    `office_address` AS "Edificio"
FROM 
    teachers
GROUP BY `office_address`;


3.
SELECT 	`exam_id` AS `Esame`, 
		AVG(`vote`,2) AS `Media Voti`
FROM 
    exam_student
GROUP BY `Esame`;


4.
SELECT	
    `department_id` AS `Dipartimento`,
	COUNT(`id`) AS `Numero di corsi di laurea`
FROM 
    degrees
GROUP BY `Dipartimento`;
